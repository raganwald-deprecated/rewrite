      def process_inner(exp)
        if exp.is_a? Array
          exp = exp.map { |sub_exp| process_inner(sub_exp) }
        end
        if unfolded = from_unfolder.unfold(exp) 
          if refolded = to_folder.fold(unfolded)
            exp = refolded
          end
        end
        exp
      end
