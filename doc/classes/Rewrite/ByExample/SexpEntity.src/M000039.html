      def unfold (sexp)
        if sexp.kind_of? Array
          self.sequence.unfolders_by_length(sexp.length).each { |unfolder|
            unfolded = unfolder.call(sexp)
            return unfolded if unfolded && (predicate.nil? || predicate.call(unfolded))
          }
          nil
        end
      end
