      def self.symbol_like_expression_matcher
          @symbol_like_expression_matcher ||= 
            returning(
              quietly do
                self.from_object(
                  s(
                    UnionOfEntitiesSequence.new(:gvar, :dvar, :vcall, :lcall, :lit, :dasgn_curr),
                    Bind.new(:variable_symbol, AnyEntity.new)
                  )
                ) 
              end
            ) do |slem|
              p slem.to_s if self.debug
            end
      end
