      def splatted?(proc)
        sexp = proc.to_sexp
        raise "Expected a proc" unless sexp[0] == :proc
        raise "Expected a proc with arity >= 1" if sexp[1] == nil
        arguments = sexp[1]
        arguments[0] == :masgn && arguments[1] && arguments[1].respond_to?(:first) && arguments[1].first != :array
      end
